  12 definition waitAdd_D :: 
  13   "'a StaProc_scheme ⇒ PART_ID_TYPE × PROC_ID_TYPE ⇒ 'a StaProc_scheme" where
  14   "waitAdd_D s pp = 
  15   (let cid = the (procBlngCore pp); tq = waitQue(cores_' s); wait_diff = waitDiff(rscPendProcs_' s);
  16       tq_wd = waitQ_SuspQAdd tq wait_diff cid pp  in
  17   if the(wait_diff pp) > 0 then
  18     s⦇cores_' := (cores_' s)⦇waitQue := fst tq_wd⦈,
  19       rscPendProcs_' := (rscPendProcs_' s)⦇waitDiff := snd tq_wd⦈⦈
  20   else s
  21   )"
  22 
  23 definition waitDel_D :: 
  24   "'a StaProc_scheme ⇒ PART_ID_TYPE × PROC_ID_TYPE ⇒ 'a StaProc_scheme" where
  25   "waitDel_D s pp =
  26   (let cid = the (procBlngCore pp); vpwl = waitQue (cores_' s); 
  27     pd = waitDiff(rscPendProcs_' s); new_vp_pd = waitQ_SuspQDel vpwl pd cid pp  in
  28   s⦇cores_' := (cores_' s)⦇waitQue := fst new_vp_pd⦈,
  29     rscPendProcs_' := (rscPendProcs_' s)⦇waitDiff := snd new_vp_pd⦈⦈)"

  31 end
