  14 lemma GET_PARTITION_STATUS :
  15   "let
  16      g_p_s = GET_PARTITION_STATUS s core_id;
  17      cur_part_id = curPartID s core_id
  18    in
  19     if cur_part_id ≠ None then
  20       g_p_s = (Some (getPartStatus s (the cur_part_id)), NO_ERROR)
  21     else
  22       g_p_s = (None, INVALID_MODE)"
  23 proof-
  24   let ?g_p_s = "GET_PARTITION_STATUS s core_id"
  25   let ?cur_part_id = "curPartID s core_id"
  26   have a10 : "?cur_part_id ≠ None ⟶ ?g_p_s = (Some (getPartStatus s (the ?cur_part_id)), NO_ERROR)"
  27     unfolding GET_PARTITION_STATUS_def by auto
  28   moreover have a11 : "?cur_part_id = None ⟶ ?g_p_s = (None, INVALID_MODE)"
  29     unfolding GET_PARTITION_STATUS_def by auto
  30   ultimately show ?thesis
  31     by metis
  32 qed

  34 lemma getPartMode_setPartMode_NotNone : "getPartMode (setPartMode s part_id md) part_id ≠ None"
  35   unfolding getPartMode_def setPartMode_def by auto
  36 
  37 lemma SET_PARTITION_MODE :
  38   "let
  39       s_p_m = SET_PARTITION_MODE s core_id oper_mode;
  40       cur_part_id = the (curPartID s core_id); 
  41       cur_mode = the(getPartMode s  cur_part_id);
  42       s_setMode = setPartMode s cur_part_id oper_mode;
  43       
  44       get_PartMode' = getPartMode (fst s_p_m) cur_part_id
  45    in 
  46       if cur_mode = NORMAL ∧ oper_mode = NORMAL then 
  47           s_p_m = (s, NO_ACTION)
  48       else if oper_mode = WARM_START ∧ cur_mode = WARM_START then 
  49           s_p_m = (s, INVALID_MODE)
  50       else if oper_mode = IDLE then 
  51           s_p_m = (s_setMode, NO_ERROR) ∧ get_PartMode' ≠ None
  52       else if oper_mode = COLD_START ∨ oper_mode = COLD_START then 
  53           s_p_m = (partRestart s_setMode core_id, NO_ERROR) 
  54       else                   
  55           s_p_m = (set_part_mode_normal s_setMode core_id, NO_ERROR)
  56   "
  57 proof-
  58   let ?s_p_m = "SET_PARTITION_MODE s core_id oper_mode"
  59   let ?cur_part_id = "the (curPartID s core_id)" 
  60   let ?cur_mode = "the(getPartMode s  ?cur_part_id)"
  61   let ?s_setMode = "setPartMode s ?cur_part_id oper_mode"
  62   let ?get_PartMode' = "getPartMode (fst ?s_p_m) ?cur_part_id"
  63   have a10 : "?cur_mode = NORMAL ∧ oper_mode = NORMAL ⟶ ?s_p_m = (s, NO_ACTION)"
  64     unfolding SET_PARTITION_MODE_def by auto
  65   moreover have a11 : "¬(?cur_mode = NORMAL ∧ oper_mode = NORMAL)
  66     ∧ (oper_mode = WARM_START ∧ ?cur_mode = WARM_START)
  67     ⟶ ?s_p_m = (s, INVALID_MODE)" unfolding SET_PARTITION_MODE_def by auto
  68   moreover have a12 : "¬(?cur_mode = NORMAL ∧ oper_mode = NORMAL)
  69     ∧ ¬(oper_mode = WARM_START ∧ ?cur_mode = WARM_START) ∧ oper_mode = IDLE 
  70     ⟶ ?s_p_m = (?s_setMode, NO_ERROR) ∧ ?get_PartMode' ≠ None"
  71   proof-
  72     {assume a12_0 : "¬(?cur_mode = NORMAL ∧ oper_mode = NORMAL)
  73     ∧ ¬(oper_mode = WARM_START ∧ ?cur_mode = WARM_START) ∧ oper_mode = IDLE"
  74      hence a12_1 : "?s_p_m = (?s_setMode, NO_ERROR)"
  75       unfolding SET_PARTITION_MODE_def by auto
  76      moreover have "?get_PartMode' ≠ None"
  77      proof-
  78       have a12_1_0 : "getPartMode ?s_setMode ?cur_part_id ≠ None"
  79         using getPartMode_setPartMode_NotNone by auto
  80       hence a12_1_1 : "getPartMode (fst ?s_p_m) ?cur_part_id ≠ None"
  81         by (metis (no_types, lifting) calculation fst_conv)
  82       hence a12_1_2 : "getPartMode (fst ?s_p_m) ?cur_part_id = ?get_PartMode'"
  83         by (metis (full_types))
  84       hence "?get_PartMode' ≠ None"
  85         by (metis (no_types, lifting) a12_1_1)
  86       thus ?thesis by (metis (no_types, lifting)) 
  87     qed
  88     ultimately have "?s_p_m = (?s_setMode, NO_ERROR) ∧ ?get_PartMode' ≠ None"
  89       by (metis (no_types, lifting))
  90     }
  91     thus ?thesis
  92       by (metis (no_types, lifting)) 
  93   qed
  94   moreover have a13 : "¬(?cur_mode = NORMAL ∧ oper_mode = NORMAL)
  95     ∧ ¬(oper_mode = WARM_START ∧ ?cur_mode = WARM_START) ∧ oper_mode ≠ IDLE 
  96     ∧ oper_mode = COLD_START ∨ oper_mode = COLD_START 
  97     ⟶ ?s_p_m = (partRestart ?s_setMode core_id, NO_ERROR)"
  98       unfolding SET_PARTITION_MODE_def by auto
  99   moreover have a14 : "¬(?cur_mode = NORMAL ∧ oper_mode = NORMAL)
 100     ∧ ¬(oper_mode = WARM_START ∧ ?cur_mode = WARM_START) ∧ oper_mode ≠ IDLE 
 101     ∧ ¬(oper_mode = COLD_START ∨ oper_mode = COLD_START)
 102     ⟶ ?s_p_m = (set_part_mode_normal ?s_setMode core_id, NO_ERROR)"
 103       unfolding SET_PARTITION_MODE_def by auto
 104   ultimately show ?thesis
 105     by (smt (z3)) 
 106 qed
